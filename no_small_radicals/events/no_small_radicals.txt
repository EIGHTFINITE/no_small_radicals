namespace = no_small_radicals

no_small_radicals.1 = {
	type = country_event
	hidden = yes
	immediate = {
		if = {
			limit = {
				NOT = { has_variable = no_small_radicals_active }
			}
			set_variable = no_small_radicals_active
		}
		if = {
			limit = {
				is_ai = yes
				is_country_alive = yes
				OR = {
					has_variable = civil_war_type_var
					is_country_type = decentralized
					country_rank <= rank_value:insignificant_power
					is_revolutionary = yes
					AND = {
						is_subject = yes
						any_scope_state = {
							is_split_state = no
							count <= 5
						}
					}
					any_scope_state = {
						is_split_state = no
						count <= 3
					}
				}
			}
			add_loyalists = {
				value = 1.0
			}
			add_loyalists = {
				value = 1.0
			}
			if = {
				limit = {
					NOT = { has_modifier = modifier_no_small_radicals }
				}
				add_modifier = modifier_no_small_radicals
			}
		}
		else = {
			if = {
				limit = {
					has_modifier = modifier_no_small_radicals
				}
				remove_modifier = modifier_no_small_radicals
				add_loyalists = {
					value = 1.0
				}
				add_loyalists = {
					value = 1.0
				}
				add_radicals = {
					value = 1.0
				}
			}
		}
		trigger_event = {
			id = no_small_radicals.1
			days = 1
		}
	}
}
